  const complexidade = Math.min(totalDirs / 20, 3); // mximo 3x

    // Penalidades mais brandas para projetos grandes
    PENALIDADE_MISSING_REQUIRED_ADAPTADO: Math.max(
      PENALIDADE_MISSING_REQUIRED * (1 - tamanhoProjeto * 0.1),

}
